문서 정보를 제어하는 문서 객체(Document Object Model)

문서 정보를 제어하는 Document객체

 

문서 정보를 표시해주는 속성들

document.title : 문서의 제목을 표시한다
document.location : 문서의 위치 경로를 표시한다

 

사용예) document 객체 사용예

<head>
    <title> Test </title>
    <script>
        document.write("이 홈페이지의 제목은", document.title,"입니다.", "<br />")
        document.write("이 홈페이지의 위치는", document.location, "입니다.")
    </script>
</head>

 

 

최종 수정 날짜를 표시해주는 속성

사용예) document.lastModified 사용예

<head>
    <title>연습</title>
    <script>
        function modifi()
        {
            a=new Date(document.lastModified)
            x=a.getYear()
            y=a.getMonth()+1
            z=a.getDate()
            document.write("이 홈페이지의 최종 수정 날짜는", x, "년", y, "월", z, "일입니다.")
        }
    </script>
</head>
<body onload="modifi()">
</body>

 

 

선택자

id 선택자(getElementById) | 요소 선택자(getElementsByTagName)

parentNode : 선택된 요소 기준으로 부모 요소를 선택할 때 사용

childNodes : 선택된 요소 기준으로 자식 요소를 선택할때 사용

children : 선택된 요소 기준으로 자식 요소를 선택할 때 사용

firstChild : 선택된 요소 기준으로 첫 번째 자식 요소를 선택할 때 사용

lastChild : 선택된 요소 기준으로 마지막 번째 자식 요소를 선택할 때 사용

previousSibling : 선택된 요소 기준으로 형제 요소 중 이전 요소를 선택할 때 사용

nextSibling : 선택된 요소 기준으로 형제 요소 중 다음 요소를 선택할 때 사용

tagName(속성) : 선택된 요소의 태그명를 반환(텍스트 노드의 속성)

nodeValue(속성) : 선택된 요소의 텍스트를 반환(텍스트 노드의 속성)

 

 

문서 색상 정보를 제어하는 속성들

document.bgColor="색깔" : 문서의 배경색을 설정
document.fgColor="색깔" : 문서의 글자색을 설정
document.linkColor="색깔" : 링크가 설정된 글자의 색을 설정
document.alinkColor="색깔" : 링크가 실행될 때의 글자색을 설정
document.vlinkColor="색깔" : 링크가 실행된 후의 글자의 색을 설정

문서 객체 조작

- 문서를 새로운 요소를 직접 생성하여 추가, 삭제, 복제하는 방법

생성 createElement("요소명") 새 요소 생성
appendTextNode /createTextNode("텍스트") 새 텍스트 생성
추가 선택요소.appendChild(새 요소) '선택 요소'에 새로운 자식 요소를 추가
선택요소1.insertBefore(새 요소, 선택 요소2) '선택 요소1'의 자식인 '선택 요소2' 앞에 새로운 자식 요소를 추가
교체 선택요소1.replaceChild(새 요소, 선택 요소2) '선택 요소1'의 자식인 '선택 요소2'를 새 요소로 덮어 씌움
삭제 선택요소1.removeChild(선택 요소2) '선택 요소1'의 자식인 '선택 요소2'를 삭제
복제 선택요소1.cloneNode(true or false) '선택 요소'를 복제하여 true인 경우에는 하위 요소까지 모두 복제
속성 부여 선택요소.setAttribute("속성","값") '선택 요소'에 해당 속성값 부여

 

사용예) createElement, setAttribute, appendChild, getElementById, appendChild 사용법

<head>
    <script>
    window.onload=function(){
        //요소 객체 생성
        var newPtag=document.createElement("p")
        var newButton1=document.createElement("button")
        var newButton2=document.createElement("button")
        var text1=document.createTextNode("버튼1")
        var text2=document.createTextNode("버튼2")

        //속성 생성
        newButton1.setAttribute("id","btn1")
        newButton2.setAttribute("id","btn2")

        //문서 객체 구조화
        newButton1.appendChild(text1)
        newButton2.appendChild(text2)
        newPtag.appendChild(newButton1)
        newPtag.appendChild(newButton2)

        //선택 요소에 새 요소 추가
        var theObj=document.getElementById("theBox")
        theObj.appendChild(newPtag)
    }
    </script>
</head>
<body>
    <div id="theBox">
        <h1>요소 생성하기</h1>
    </div>
</body>

 

사용예) insertBefore, removeChild, cloneNOde 사용법

<head>
    <script>
    window.onload=function(){
        var theList=document.getElementById("theList")
        var listAll=theList.getElementsByTagName("li")

        //리스트 1을 복제합니다.
        var copyList=listAll[3].cloneNode(true)

        //복제된 리스트1를 리스트2앞으로 삽입합니다.
        theList.insertBefore(copyList, listAll[0])
    }
    //기존 리스트1을 삭제합니다.
    theList.removeChild(listAll[4])
    </script>
</head>
<body>
    <h1>요소 생성하기</h1>
    <ul id="theList">
        <li>리스트2</li>
        <li>리스트3</li>
        <li>리스트4</li>
        <li>리스트1</li>
    </ul>
</body>
